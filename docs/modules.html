useResetableRef

  • useResetableRef<T>(initialValue: MaybeRef<T>): [Ref<T>, (value?: MaybeRef<T>) => void]
    • Defined in vhooks/src/state/resetableRef.ts:29

    A hooks create resetable ref

    Example:

    import { useResetableRef } from '@m9ch/vhooks'
    import { defineComponent } from 'vue'

    const [count, reset] = useResetableRef(0)

    const App = defineComponent(() => {
      onMounted(() => {
        count.value++
      })

      onBeforeUnmount(() => {
        reset()
      })

      return () => <div>{count.value}</div>
    })
